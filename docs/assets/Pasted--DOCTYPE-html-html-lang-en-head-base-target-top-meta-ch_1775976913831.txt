      document.querySelectorAll('.nav-tab').forEach(t => {
        t.classList.remove('active');
        t.setAttribute('aria-selected', 'false');

      btn.classList.add('active');

        const start       = document.getElementById('man-start').value;
        const expiry      = document.getElementById('man-expiry').value;
        const price       = document.getElementById('man-price').value;
        const fee         = document.getElementById('man-fee').value;

        need(asset,       'Asset Name', missing);

        need(start,       'Start Date', missing);
        need(expiry,      'Expiry Date', missing);
        need(broker,      'Handling Broker', missing);
        needNum(price,    'Asking Price (USD)', missing);
        needNum(fee,      'Fee % (decimal)', missing);
        data = {
          assetName:           asset,
          assetType:           assetType,

          mandateType:         mandateType,

      setTimeout(() => { SUB.textContent = 'Writing to database…'; }, 500);

        .withSuccessHandler(result => {
          showSuccess(result.message || 'Record saved');
          setTimeout(() => {
            hideOverlay();
            clearPanel(type);
            if (typeof google !== 'undefined' && google.script.host) {
              google.script.host.close();
            }
          }, 1800);
        })
        .withFailureHandler(err => {
          showError(err.message || 'Unknown error — please try again');
          setTimeout(hideOverlay, 3500);
        })
        [fn](data);
    }
